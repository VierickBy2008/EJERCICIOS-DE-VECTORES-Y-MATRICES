#include <iostream>
using namespace std;

int main() {
    char palabra[100]; 
    int contador = 0;

    cout << "Ingrese una palabrita: ";
    cin >> palabra;
    while (palabra[contador] != '\0') {
        contador++;
    }

    cout << "La frase tiene " << contador << " letras o caracteres." << endl;

    return 0;
}
