#include <iostream>
using namespace std;

int main() {
    int n, m;
    float matriz[20][20], suma_principal = 0, suma_inversa = 0;

    cout << "Tamano de filas (n): "; 
    cin >> n;
    
    cout << "Tamano de columnas (m): "; 
    cin >> m;

    cout << "\nIngrese los " << n * m << " elementos de la matriz:" << endl;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << "Elemento [" << i << "][" << j << "]: ";
            cin >> matriz[i][j];
        }
    }

    cout << "\nMatriz ingresada:" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << matriz[i][j] << "\t";
        }
        cout << endl; 
    }
    cout << endl;

    int limite;
    if (n < m) {
        limite = n; // Si hay menos filas que columnas, el límite es n
    } else {
        limite = m; // En caso contrario (hay menos o igual cantidad de columnas), el límite es m
    }

    for (int i = 0; i < limite; i++) {
        suma_principal += matriz[i][i];                
        suma_inversa += matriz[n - 1 - i][m - 1 - i];         
    }

    cout << "La suma de la diagonal normal es: " << suma_principal << endl;
    cout << "La suma de la diagonal desde la otra punta es: " << suma_inversa << endl;

    return 0;
}
