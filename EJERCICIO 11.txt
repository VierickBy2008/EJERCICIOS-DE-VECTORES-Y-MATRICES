#include <iostream>
using namespace std;

int main() {
    int n, m;
    float matriz[20][20], suma_principal = 0, suma_inversa = 0;

    cout << "Tamano de filas (n): "; 
    cin >> n;
    
    cout << "Tamano de columnas (m): "; 
    cin >> m;

    cout << "Ingrese los " << n * m << " elementos de la matriz (separados por espacio o Enter):" << endl;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << "Elemento [" << i << "][" << j << "]: ";
            cin >> matriz[i][j];
        }
    }

    cout << "\nMatriz ingresada:" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << matriz[i][j] << "\t";
        }
        cout << endl; 
    }
    cout << endl;

    if (n == m) {
        for (int i = 0; i < n; i++) {
            suma_principal += matriz[i][i];                
            suma_inversa += matriz[i][n - 1 - i];         
        }

        cout << "La suma de la diagonal principal es: " << suma_principal << endl;
        cout << "La suma de la diagonal inversa es: " << suma_inversa << endl;
    } else {
        cout << "Advertencia: La matriz no es cuadrada (" << n << "x" << m << ")." << endl;
        cout << "Las diagonales principal e inversa solo se definen en matrices cuadradas (n = m)." << endl;
    }

    return 0;
}
